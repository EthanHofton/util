517 std::string format(const std::string& t_sink, const std::string& t_msg, const LOGGER_LEVEL& t_level)
518 {
519 std::string result = m_sinks[t_sink]->getFormat();
520 size_t pos = result.find("%");
521 while (pos != std::string::npos)
522 {
523 if (pos + 1 > result.size())
524 {
525 throw std::invalid_argument("no arg provided");
526 }
527
528 char command = result[pos+1];
529 std::string replValue = "";
530 switch(command)
531 {
532 case 'v': replValue = t_msg; break;
533 case 't': { std::ostringstream ss; ss << std::this_thread::get_id(); replValue = ss.str(); } break;
534 case 'P': replValue = std::to_string(getpid()); break;
535 case 'n': replValue = t_sink; break;
536 case 'l': { std::stringstream ss; ss << t_level; replValue = ss.str(); } break;
537 case 'g': replValue = __FILE__; break;
538 case 'E': replValue = std::to_string(std::time(0)); break;
539 case 'a': case 'A': case 'b': case 'B': case 'c': case 'm':
540 case 'H': case 'I': case 'M': case 'S': case 'p': case 'r':
541 case 'T': case 'X': case 'z': case 'R': case 'd':
542 replValue = strfmttime(std::string("%") + command); break;
543 case 'C': replValue = strfmttime("%g"); break;
544 case 'Y': replValue = strfmttime("%G"); break;
545 case 'D': case 'x': replValue = strfmttime("%D"); break;
546 case '^': replValue = m_sinks[t_sink]->startOutputColorRange(t_level); break;
547 case '$': replValue = m_sinks[t_sink]->endOutputColorRange(); break;
548 }
549
550 result.replace(pos, 2, replValue);
551 pos = result.find("%");
552 }
553
554 return result;
555 }
