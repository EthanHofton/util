518 std::string format(const std::string& t_sink, const std::string& t_msg, const LOGGER_LEVEL& t_level)
519 {
520 std::string result = m_sinks[t_sink]->getFormat();
521 size_t pos = result.find("%");
522 while (pos != std::string::npos)
523 {
524 if (pos + 1 > result.size())
525 {
526 throw std::invalid_argument("no arg provided");
527 }
528
529 char command = result[pos+1];
530 std::string replValue = "";
531 switch(command)
532 {
533 case 'v': replValue = t_msg; break;
534 case 't': { std::ostringstream ss; ss << std::this_thread::get_id(); replValue = ss.str(); } break;
535 case 'P': replValue = std::to_string(getpid()); break;
536 case 'n': replValue = t_sink; break;
537 case 'l': { std::stringstream ss; ss << t_level; replValue = ss.str(); } break;
538 case 'g': replValue = __FILE__; break;
539 case 'E': replValue = std::to_string(std::time(0)); break;
540 case 'a': case 'A': case 'b': case 'B': case 'c': case 'm':
541 case 'H': case 'I': case 'M': case 'S': case 'p': case 'r':
542 case 'T': case 'X': case 'z': case 'R': case 'd':
543 replValue = strfmttime(std::string("%") + command); break;
544 case 'C': replValue = strfmttime("%g"); break;
545 case 'Y': replValue = strfmttime("%G"); break;
546 case 'D': case 'x': replValue = strfmttime("%D"); break;
547 case '^': replValue = m_sinks[t_sink]->startOutputColorRange(t_level); break;
548 case '$': replValue = m_sinks[t_sink]->endOutputColorRange(); break;
549 }
550
551 result.replace(pos, 2, replValue);
552 pos = result.find("%");
553 }
554
555 return result;
556 }
