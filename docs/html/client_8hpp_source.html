342 void connectServer() {
343 // * only connect if the server socket has been closed
344 if (m_closed == true) {
345 // * create server socket and check for error
346 if ((m_serverFd = socket(AF_INET, SOCK_STREAM, 0)) < 0) {
347 logMessage(util::LOGGER_LEVEL::CRITICAL, "failed to create server socket");
348 logMessage(util::LOGGER_LEVEL::ERROR, util::fmt("socket error: {}", strerror(errno)));
349 throw std::runtime_error("failed to create server socket");
350 }
351 m_closed = false;
352
353 // * define server port and address
354 sockaddr_in serveraddr;
355 serveraddr.sin_port = htons(m_port);
356 serveraddr.sin_family = AF_INET;
357
358 // * conver IPv4 to IPv6 and set in sockaddr_in struct and check for error
359 if (inet_pton(AF_INET, m_ip.c_str(), &serveraddr.sin_addr) <= 0) {
360 logMessage(util::LOGGER_LEVEL::CRITICAL, "server IP address not valid");
361 logMessage(util::LOGGER_LEVEL::ERROR, util::fmt("inet_pron error: {}", strerror(errno)));
362 throw std::runtime_error("server IP address not valid");
363 }
364
365 // * log that the clinet is about to connect to server
366 logMessage(util::LOGGER_LEVEL::INFO, util::fmt("connecting to server at ip {} on port {}", m_ip, m_port));
367
368 // * connect to server
369 if ((m_clientFd = connect(m_serverFd, (sockaddr*)&serveraddr, sizeof(serveraddr))) < 0) {
370 logMessage(util::LOGGER_LEVEL::CRITICAL, "failed to connect to server");
371 logMessage(util::LOGGER_LEVEL::ERROR, util::fmt("connect error: {}", strerror(errno)));
372 throw std::runtime_error("failed to connect to server");
373 }
374
375 // * server connected evnet and dispatch
376 client_connect_event connEv(m_serverFd, m_clientFd, m_ip, m_port);
377 m_eventFunction(connEv);
378
379 // * log client is connected
380 logMessage(util::LOGGER_LEVEL::INFO, util::fmt("connected to server at ip {} on port {}", m_ip, m_port));
381 }

402 void clientListen() {
403 client_start_listen_event startListenEv;
404 m_eventFunction(startListenEv);
405 logMessage(util::LOGGER_LEVEL::INFO, "client listining for activity");
406
407 // * setup file discriptor set and timeout
408 fd_set readfds;
409 timeval tv;
410 tv.tv_sec = 3;
411 tv.tv_usec = 50;
412
413 while (m_alive) {
414 // * clear the file discriptor set
415 FD_ZERO(&readfds);
416 // * add the server socket to the file discriptor set
417 FD_SET(m_serverFd, &readfds);
418
419 // * get the activity of the set
420 int activity = select(m_serverFd + 1, &readfds, NULL, NULL, &tv);
421
422 // * negative activity means error, deal with accordingly
423 if (activity < 0) {
424
425 }
426
427 // * zero activity means timeout (or no activity)
428 if (activity == 0) {
429 continue;
430 }
431
432 // * check for activity of the socket
433 if (FD_ISSET(m_serverFd, &readfds)) {
434 if (handleServerActivity()) {
435 m_alive = false;
436 continue;
437 }
438 }
439 }
440
441 client_finished_listen_event finishedListenEv;
442 m_eventFunction(finishedListenEv);
443 logMessage(util::LOGGER_LEVEL::INFO, "client stopped listinig for activity");
