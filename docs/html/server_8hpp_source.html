390 server(const int& t_port, const SERVER_METHOD& t_serverMethod, eventFn t_eventFn, util::logger& t_serverLogger) : m_serverLogger(t_serverLogger) {
391 // * set the port, server method and event function
392 m_port = t_port;
393 m_serverMethod = t_serverMethod;
394 m_eventFunction = t_eventFn;
395
396 // * initalize listening to false
397 m_listening = false;
398
399 // * pre-declare vars
400 sockaddr_in address;
401 int addrlen = sizeof(address);
402 int opt = 1;
403
404 // * create main server socket
405 if ((m_serverFd = socket(AF_INET, SOCK_STREAM, 0)) < 0) {
406 logMessage(util::LOGGER_LEVEL::CRITICAL, "failed to create server socket");
407 logMessage(util::LOGGER_LEVEL::ERROR, util::fmt("socket error: {}", strerror(errno)));
408 throw std::runtime_error("failed to create server socket");
409 }
410
411 // * set socket options
412 if (setsockopt(m_serverFd, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt)) < 0) {
413 logMessage(util::LOGGER_LEVEL::CRITICAL, "failed to set socket options");
414 logMessage(util::LOGGER_LEVEL::ERROR, util::fmt("setsockopt error: {}", strerror(errno)));
415 throw std::runtime_error("failed to set socket options");
416 }
417
418 // * set the socket ports and adresses
419 address.sin_family = AF_INET;
420 address.sin_port = htons(m_port);
421 address.sin_addr.s_addr = INADDR_ANY;
422
423 // * bind the server to the addresses and ports
424 if (bind(m_serverFd, (sockaddr*)&address, addrlen) < 0) {
425 logMessage(util::LOGGER_LEVEL::CRITICAL, "failed to bind server socket");
426 logMessage(util::LOGGER_LEVEL::ERROR, util::fmt("bind error: {}", strerror(errno)));
427 throw std::runtime_error("failed to bind server socket");
428 }
429
430 // * mark the server socket as set to listen
431 if (listen(m_serverFd, 3) < 0) {
432 logMessage(util::LOGGER_LEVEL::CRITICAL, "failed to set server socket as listening");
433 logMessage(util::LOGGER_LEVEL::ERROR, util::fmt("listen error: {}", strerror(errno)));
434 throw std::runtime_error("failed to set server socket as listening");
435 }
436
437 // * log the the server was properly initalized
438 logMessage(util::LOGGER_LEVEL::INFO, util::fmt("initalized server on port {}", m_port));
439
440 // * create a server startup event and dispactch
441 server_startup_event startupEv(m_serverFd, m_port, m_serverMethod);
442 m_eventFunction(startupEv);
443
444 // * set the listening flag
445 m_listening = true;
446 m_listenThread = std::async(std::launch::async, &server::serverListen, this);
447 }

574 void serverListen() {
575 // * log server listen
576 logMessage(util::LOGGER_LEVEL::INFO, "server now listening for connections");
577
578 // * setup file discriptor set and timeout
579 fd_set readfds;
580 timeval tv;
581 tv.tv_sec = 3;
582 tv.tv_usec = 50;
583
584 while (m_listening) {
585 // * clear the socket discriptor set
586 FD_ZERO(&readfds);
587
588 // * add the server socket to the fd set
589 FD_SET(m_serverFd, &readfds);
590 int max_sd = m_serverFd;
591
592 for (auto &con : m_clientMap) {
593 int sd = con.second;
594
595 // * if the client socket descriptor is valid, add it to the set
596 if (sd > 0) {
597 FD_SET(sd, &readfds);
598 }
599
600 // * get the max sokcet discriptor (for select)
601 max_sd = std::max(max_sd, sd);
602 }
603
604 // * get the activity of the socket discriptors
605 int activity = select(max_sd + 1, &readfds, NULL, NULL, &tv);
606
607
608 // * negative activity is an error, deal with accordingly
609 if (activity < 0) {
610
611 }
612
613 // * 0 activity is timeout (or no activity)
614 if (activity == 0) {
615 continue;
616 }
617
618 // * if the master socket is set, new incoming connection
619 if (FD_ISSET(m_serverFd, &readfds)) {
620 connectClient();
621 }
622
623 // * if the client sockets are set, incoming message or client disconnection
624 // * loop through all the clients and check if they are set
625 std::vector<std::string> disconnectedClients;
626 for (auto& clientCon : m_clientMap) {
627 int sd = clientCon.second;
628
629 if (FD_ISSET(sd, &readfds)) {
630 // * check if a client has disconnected from the server
631 if (handleClient(clientCon.first)) {
632 // * if client disconnected, add to list of disconnecrted clients
633 disconnectedClients.push_back(clientCon.first);
634 continue;
635 }
636 }
637 }
638
639 for (auto& clientId : disconnectedClients) {
640 disconnectClient(clientId);
641 }
642 }
643
644 logMessage(util::LOGGER_LEVEL::INFO, "server stopped listening for connections");
645 }
