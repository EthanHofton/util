200 server(const int& t_port, const SERVER_METHOD& t_serverMethod, eventFn t_eventFn, util::logger& t_serverLogger) : m_serverLogger(t_serverLogger) {
201 // * set the port, server method and event function
202 m_port = t_port;
203 m_serverMethod = t_serverMethod;
204 m_eventFunction = t_eventFn;
205
206 // * initalize listening to false
207 m_listening = false;
208
209 // * pre-declare vars
210 sockaddr_in address;
211 int addrlen = sizeof(address);
212 int opt = 1;
213
214 // * create main server socket
215 if ((m_serverFd = socket(AF_INET, SOCK_STREAM, 0)) < 0) {
216 logMessage(util::LOGGER_LEVEL::CRITICAL, "failed to create server socket");
217 logMessage(util::LOGGER_LEVEL::ERROR, util::fmt("socket error: {}", strerror(errno)));
218 throw std::runtime_error("failed to create server socket");
219 }
220
221 // * set socket options
222 if (setsockopt(m_serverFd, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt)) < 0) {
223 logMessage(util::LOGGER_LEVEL::CRITICAL, "failed to set socket options");
224 logMessage(util::LOGGER_LEVEL::ERROR, util::fmt("setsockopt error: {}", strerror(errno)));
225 throw std::runtime_error("failed to set socket options");
226 }
227
228 // * set the socket ports and adresses
229 address.sin_family = AF_INET;
230 address.sin_port = htons(m_port);
231 address.sin_addr.s_addr = INADDR_ANY;
232
233 // * bind the server to the addresses and ports
234 if (bind(m_serverFd, (sockaddr*)&address, addrlen) < 0) {
235 logMessage(util::LOGGER_LEVEL::CRITICAL, "failed to bind server socket");
236 logMessage(util::LOGGER_LEVEL::ERROR, util::fmt("bind error: {}", strerror(errno)));
237 throw std::runtime_error("failed to bind server socket");
238 }
239
240 // * mark the server socket as set to listen
241 if (listen(m_serverFd, 3) < 0) {
242 logMessage(util::LOGGER_LEVEL::CRITICAL, "failed to set server socket as listening");
243 logMessage(util::LOGGER_LEVEL::ERROR, util::fmt("listen error: {}", strerror(errno)));
244 throw std::runtime_error("failed to set server socket as listening");
245 }
246
247 // * log the the server was properly initalized
248 logMessage(util::LOGGER_LEVEL::INFO, util::fmt("initalized server on port {}", m_port));
249
250 // * create a server startup event and dispactch
251 server_startup_event startupEv(m_serverFd, m_port, m_serverMethod);
252 m_eventFunction(startupEv);
253
254 // * set the listening flag
255 m_listening = true;
256 m_listenThread = std::async(std::launch::async, &server::serverListen, this);
257 }

344 void serverListen() {
345 // * log server listen
346 logMessage(util::LOGGER_LEVEL::INFO, "server now listening for connections");
347
348 // * setup file discriptor set and timeout
349 fd_set readfds;
350 timeval tv;
351 tv.tv_sec = 3;
352 tv.tv_usec = 50;
353
354 while (m_listening) {
355 // * clear the socket discriptor set
356 FD_ZERO(&readfds);
357
358 // * add the server socket to the fd set
359 FD_SET(m_serverFd, &readfds);
360 int max_sd = m_serverFd;
361
362 for (auto &con : m_clientMap) {
363 int sd = con.second;
364
365 // * if the client socket descriptor is valid, add it to the set
366 if (sd > 0) {
367 FD_SET(sd, &readfds);
368 }
369
370 // * get the max sokcet discriptor (for select)
371 max_sd = std::max(max_sd, sd);
372 }
373
374 // * get the activity of the socket discriptors
375 int activity = select(max_sd + 1, &readfds, NULL, NULL, &tv);
376
377
378 // * negative activity is an error, deal with accordingly
379 if (activity < 0) {
380
381 }
382
383 // * 0 activity is timeout (or no activity)
384 if (activity == 0) {
385 continue;
386 }
387
388 // * if the master socket is set, new incoming connection
389 if (FD_ISSET(m_serverFd, &readfds)) {
390 connectClient();
391 }
392
393 // * if the client sockets are set, incoming message or client disconnection
394 // * loop through all the clients and check if they are set
395 std::vector<std::string> disconnectedClients;
396 for (auto& clientCon : m_clientMap) {
397 int sd = clientCon.second;
398
399 if (FD_ISSET(sd, &readfds)) {
400 // * check if a client has disconnected from the server
401 if (handleClient(clientCon.first)) {
402 // * if client disconnected, add to list of disconnecrted clients
403 disconnectedClients.push_back(clientCon.first);
404 continue;
405 }
406 }
407 }
408
409 for (auto& clientId : disconnectedClients) {
410 disconnectClient(clientId);
411 }
412 }
413
414 logMessage(util::LOGGER_LEVEL::INFO, "server stopped listening for connections");
415 }
